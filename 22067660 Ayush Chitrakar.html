import numpy as np
from scipy.stats import skew, kurtosis
def sum_stats(data, variable):
    #calculating the sum of the variable
    total_sum = data[variable].sum()

    #calculating the mean of the variable
    mean_value = data[variable].mean()

    #calculating the standard deviation of the variable
    standard_dev = data[variable].std()

    #calculating the skewness of the variable
    skewness = skew(data[variable])

    #calculating the kurtosis of the variable
    kurtos = kurtosis(data[variable])

    #displaying summary of the statistics
    print("Summary statistics for variable:", variable)
    print("Sum:", total_sum)
    print("Mean:", mean_value)
    print("Standard Deviation:", standard_dev)
    print("Skewness:", skewness)
    print("Kurtosis:", kurtos)

#choosing a variable
variable = 'salary_in_usd'

#calculating and displaying the summary statistics
